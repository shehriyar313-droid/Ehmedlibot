import asyncio
import logging
from aiogram import Bot, Dispatcher
from aiogram.types import BotCommand, Message
from aiogram.filters import Command
from config import BOT_TOKEN

logging.basicConfig(level=logging.INFO)

bot = Bot(token=BOT_TOKEN)
dp = Dispatcher()

@dp.message(Command("start"))
async def start_command(message: Message):
    text = """
🤖 **EhmedliBot** - Məlumat Axtarış Botu

🔍 **Mövcud Komandalar:**

📞 /phone 0512345678 - Nömrə məlumatı
💻 /ip 1.1.1.1 - IP ünvan məlumatı

💎 /premium - Limitsiz axtarış (30 AZN/ay)
"""
    await message.answer(text, parse_mode="Markdown")

@dp.message(Command("help"))

    text = """
📚 **Kömək**

/phone 0512345678 - Operator, region
/ip 1.1.1.1 - IP məlumatı
/premium - Premium almaq

**Pulsuz limit:** gündə 3 axtarış
"""
    await message.answer(text, parse_mode="Markdown")

@dp.message(Command("phone"))
async def phone_search(message: Message):
    import re
    args = message.text.split()
    if len(args) < 2:
        await message.answer("❌ /phone 0512345678")
        return
    
    phone = args[1]
    phone = re.sub(r'\D', '', phone)
    if len(phone) == 10:
        phone = "994" + phone
    
    operators = {
        '50': 'Azercell', '51': 'Azercell',
        '55': 'Bakcell', '70': 'Bakcell', '77': 'Bakcell',
        '99': 'Nar', '10': 'Nar', '60': 'Nar'
    }
    
    prefix = phone[2:4] if len(phone) == 12 else phone[:2]
    operator = operators.get(prefix, 'Bilinməyən')
    
    result = f"📞 **Nömrə Məlumatı**\n\n"
    result += f"📱 Nömrə: +{phone}\n"
    result += f"📡 Operator: {operator}\n"
    
    await message.answer(result, parse_mode="Markdown")

@dp.message(Command("ip"))
async def ip_search(message: Message):
    import aiohttp
    args = message.text.split()
    if len(args) < 2:
        await message.answer("❌ /ip 1.1.1.1")
        return
    
    ip = args[1]
    await message.answer("🔍 IP yoxlanılır...")
    
    async with aiohttp.ClientSession() as session:
        try:
            async with session.get(f"http://ip-api.com/json/{ip}") as resp:
                data = await resp.json()
                if data['status'] == 'success':
                    result = f"🌍 **IP Məlumatı**\n\n"
                    result += f"🔢 IP: `{ip}`\n"
                    result += f"🇺🇳 Ölkə: {data['country']}\n"
                    result += f"🏙️ Şəhər: {data['city']}\n"
                    result += f"📡 Provayder: {data['isp']}"
                else:
                    result = f"❌ Məlumat tapılmadı: {ip}"
        except:
            result = f"❌ Xəta baş verdi"
    
    await message.answer(result, parse_mode="Markdown")

async def set_commands():
    commands = [
        BotCommand(command="start", description="🚀 Botu başlat"),
        BotCommand(command="help", description="📚 Kömək"),
        BotCommand(command="phone", description="📞 Nömrə məlumatı"),
        BotCommand(command="ip", description="💻 IP məlumatı"),
        BotCommand(command="premium", description="💎 Premium al"),
    ]
    await bot.set_my_commands(commands)

async def main():
    await set_commands()
    await bot.delete_webhook(drop_pending_updates=True)
    await dp.start_polling(bot)

if __name__ == "__main__":
    asyncio.run(main())
